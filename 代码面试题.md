
### 给定一棵树, 将树转成一位数组并排序

```js
const tree = [
    {
        value: 1,
        children: [
            {
                value: 2,
                children: [
                    {
                        value: 4
                    },
                    {
                        value: 5
                    },
                    {
                        value: 6
                    },
                ]
            },
            {
                value: 3,
                children: [
                    { value: 7 }
                ]
            }
        ]
    }
];

const convert  = (tree, arr=[]) => {
    if(!Array.isArray(tree))
        return [];
    for(node of tree){
        arr.push(node.value);
        if(Array.isArray(node.children)){
            convert(node.children, arr);
        }
    }
    return arr;
}

let arr = convert(tree);
let result = arr.sort();
console.log(result)
```
